快速入门

# 安装Vercel CLI
npm install -g vercel

# 登录Vercel
vercel login

# 部署项目
vercel

# 部署到生产环境
vercel --prod

# 查看部署状态
vercel ls

# 查看部署详情
vercel inspect [deployment-url]

# 删除部署
vercel rm [deployment-url]

配置示例

// vercel.json
{
    // 项目名称
    "name": "my-project",
    
    // 构建配置
    "builds": [
        {
            "src": "package.json",
            "use": "@vercel/node"
        }
    ],
    
    // 路由配置
    "routes": [
        {
            "src": "/api/(.*)",
            "dest": "/api/$1"
        },
        {
            "src": "/(.*)",
            "dest": "/index.html"
        }
    ],
    
    // 环境变量
    "env": {
        "DATABASE_URL": "@database-url",
        "API_KEY": "@api-key"
    },
    
    // 区域配置
    "regions": ["iad1", "sfo1", "lhr1"],
    
    // 函数配置
    "functions": {
        "api/**/*.js": {
            "memory": 1024,
            "maxDuration": 10
        }
    }
}
